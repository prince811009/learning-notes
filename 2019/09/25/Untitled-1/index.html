What's THIS in JavaScript ?

Sep 25 2019 Documentation 8 minutes read (About 1257 words)

前言

常常出現以及使用的

class Car {
setName(name) {
this.name = name
}

getName() {
    return this.name
}

}

const myCar = new Car()
myCar.setName(‘hello’)
console.log(myCar.getName()) // hello

‘use strict’;
function hello(a, b){
console.log(this, a, b)
}

hello(1, 2) // undefined 1 2
hello.call(undefined, 1, 2) // undefined 1 2
hello.apply(undefined, [1, 2]) // undefined 1 2

        'use strict';
        function hello(a, b){
            console.log(this, a, b)
        }

        hello.call('yo', 1, 2) // yo 1 2
        hello.apply('hihihi', [1, 2]) // hihihi 1 2

const obj = {
value: 1,
hello: function() {
console.log(this.value)
}
}

obj.hello() // 1

const obj = {
    value: 1,
    hello: function() {
        console.log(this.value)
    }
}

obj.hello() // 1
const hey = obj.hello
hey() // undefined
