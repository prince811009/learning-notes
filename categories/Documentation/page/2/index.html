JavaScript 中 new 指令的缺點

使用

function Dog(name) {
this.name = name
this.species = ‘犬科’;
}

var dogA = new Dog(‘Doggy’);
var dogB = new Dog(‘Lucky’);

dogA.species = ‘貓科’;
console.log(dogB.species); // 犬科

function Dog(name) {
this.name = name;
}

Dog.prototype = {
species: ‘犬科’
}

var dogA = new Dog(‘Doggy’);
var dogB = new Dog(‘Lucky’);

console.log(dogA.species); // 犬科
console.log(dogB.species); // 犬科
